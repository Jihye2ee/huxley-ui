## Usage

```tsx
import {
  AlertDialogRoot, AlertDialogTrigger, AlertDialogContent,
  AlertDialogTitle, AlertDialogDescription, AlertDialogFooter,
  AlertDialogClose, AlertDialogAction, AlertDialogCancel,
} from "@/components/ui/alert-dialog";
import { Button } from "@/components/ui/button";

export function Example() {
  return (
    <AlertDialogRoot>
      <AlertDialogTrigger render={<Button appearance="error" size="medium" />}>
        Delete
      </AlertDialogTrigger>
      <AlertDialogContent>
        <AlertDialogClose />
        <AlertDialogTitle>Are you sure?</AlertDialogTitle>
        <AlertDialogDescription>This cannot be undone.</AlertDialogDescription>
        <AlertDialogFooter>
          <AlertDialogCancel render={<Button appearance="ghost" size="medium" />}>Cancel</AlertDialogCancel>
          <AlertDialogAction render={<Button size="medium" />}>Confirm</AlertDialogAction>
        </AlertDialogFooter>
      </AlertDialogContent>
    </AlertDialogRoot>
  );
}
```

## Variants

### Two Buttons

```tsx
<AlertDialogFooter>
  <AlertDialogCancel render={<Button appearance="ghost" size="medium" />}>취소</AlertDialogCancel>
  <AlertDialogAction render={<Button size="medium" />}>확인</AlertDialogAction>
</AlertDialogFooter>
```

### Directional

```tsx
<AlertDialogFooter variant="directional">
  <AlertDialogCancel render={<Button appearance="ghost" size="medium" />}>이전</AlertDialogCancel>
  <AlertDialogAction render={<Button size="medium" />}>다음</AlertDialogAction>
</AlertDialogFooter>
```

## Props

### AlertDialogContent

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | 추가 CSS 클래스 |

### AlertDialogFooter

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `"default" \| "directional"` | `"default"` | 버튼 정렬 방식 |

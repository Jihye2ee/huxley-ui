## Usage

```tsx
import {
  ComboboxRoot, ComboboxInput, ComboboxContent,
  ComboboxItem, ComboboxEmpty,
} from "@/components/ui/combobox";

export function Example() {
  return (
    <ComboboxRoot>
      <ComboboxInput placeholder="Search..." />
      <ComboboxContent>
        <ComboboxItem value="1">Option 1</ComboboxItem>
        <ComboboxItem value="2">Option 2</ComboboxItem>
        <ComboboxEmpty />
      </ComboboxContent>
    </ComboboxRoot>
  );
}
```

## States

```tsx
<ComboboxInput placeholder="Disabled" disabled />
<ComboboxInput placeholder="Error" error />
```

## Props

### ComboboxInput

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `inputSize` | `"md" \| "sm"` | `"md"` | 입력 크기 |
| `error` | `boolean` | `false` | 에러 상태 |
| `placeholder` | `string` | - | 플레이스홀더 |
| `disabled` | `boolean` | `false` | 비활성화 |
